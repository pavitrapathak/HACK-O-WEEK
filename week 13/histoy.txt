<script>
class BrowserHistory {
  constructor(homepage) {
    this.current = homepage;
    this.backStack = [];
    this.forwardStack = [];

    this.loadHistory();
    this.updateUI();
  }

  visit(input) {
    const url = this.processInput(input);

    this.backStack.push(this.current);
    this.current = url;
    this.forwardStack = [];

    this.saveHistory();
    this.updateUI();
  }

  back() {
    if (this.backStack.length === 0) return;

    this.forwardStack.push(this.current);
    this.current = this.backStack.pop();

    this.saveHistory();
    this.updateUI();
  }

  forward() {
    if (this.forwardStack.length === 0) return;

    this.backStack.push(this.current);
    this.current = this.forwardStack.pop();

    this.saveHistory();
    this.updateUI();
  }

  processInput(input) {
    if (input.includes(".")) {
      if (!input.startsWith("http")) {
        return "https://" + input;
      }
      return input;
    } else {
      return "https://www.google.com/search?q=" + input;
    }
  }

  updateUI() {
    document.getElementById("currentPage").innerText = this.current;
    document.getElementById("viewer").src = this.current;

    document.getElementById("backStack").innerHTML =
      this.backStack.map(i => `<div class="item">${i}</div>`).join("");

    document.getElementById("forwardStack").innerHTML =
      this.forwardStack.map(i => `<div class="item">${i}</div>`).join("");
  }

  saveHistory() {
    localStorage.setItem("browserHistory", JSON.stringify({
      current: this.current,
      back: this.backStack,
      forward: this.forwardStack
    }));
  }

  loadHistory() {
    const data = JSON.parse(localStorage.getItem("browserHistory"));
    if (data) {
      this.current = data.current;
      this.backStack = data.back;
      this.forwardStack = data.forward;
    }
  }
}

const browser = new BrowserHistory("https://example.com");

function visit() {
  const input = document.getElementById("urlInput").value;
  if (input) {
    browser.visit(input);
    document.getElementById("urlInput").value = "";
  }
}

function goBack() {
  browser.back();
}

function goForward() {
  browser.forward();
}
